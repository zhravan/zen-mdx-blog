## Code Enhancements in Action

Here's a React component with all code features enabled:

```tsx filename="components/BlogPost.tsx" {5-8,15-18} showLineNumbers
import { Suspense } from 'react';
import { notFound } from 'next/navigation';
import { getAllPosts } from '@/lib/blog';

// Import plugin utilities
import { getTocForPost } from '@/lib/plugins/toc';
import { getReadingTime } from '@/lib/plugins/reading-time';
import { getSeriesForPost } from '@/lib/plugins/series';

import { TableOfContents } from '@/components/TableOfContents';
import { SeriesNavigator } from '@/components/SeriesNavigator';
import { ViewCounter } from '@/components/ViewCounter';

export default async function BlogPost({ params }) {
  // Load post data
  const { slug } = await params;
  const post = getAllPosts().find(p => p.slug === slug);
  
  if (!post) {
    notFound();
  }
  
  // Load plugin data
  const toc = await getTocForPost(slug);
  const readingTime = await getReadingTime(slug);
  const series = getSeriesForPost(slug);
  
  return (
    <div className="blog-layout">
      {series && <SeriesNavigator series={series} />}
      
      <article>
        <h1>{post.title}</h1>
        <div className="metadata">
          <time>{post.date}</time>
          {readingTime && <span>{readingTime.text}</span>}
          <ViewCounter slug={slug} />
        </div>
        
        <Content />
      </article>
      
      {toc && <TableOfContents headings={toc} />}
    </div>
  );
}
```

## Search Functionality

The search plugin provides:

```typescript filename="lib/plugins/search.ts" {8-12}
export async function searchPosts(query: string): Promise<SearchResult[]> {
  const index = await buildSearchIndex();
  const results: SearchResult[] = [];

  for (const item of index) {
    let maxScore = 0;

    // Search in title (highest weight)
    if (config.searchFields.includes('title')) {
      const score = fuzzySearch(query, item.title);
      maxScore = Math.max(maxScore, score * 1.5);
    }

    // Search in description
    if (config.searchFields.includes('description')) {
      const score = fuzzySearch(query, item.description);
      maxScore = Math.max(maxScore, score);
    }

    if (maxScore > 0) {
      results.push({ ...item, score: maxScore });
    }
  }

  return results.sort((a, b) => b.score - a.score);
}
```

Try the search (⌘K or Ctrl+K) to find posts!

## Configuration-Driven Design

All features are controlled via `config/plugins.yaml`:

```yaml
plugins:
  # Code features
  code-enhancements:
    enabled: true
    showFileName: true
    showLineNumbers: true
    highlightLines: true
    
  # Content features  
  series:
    enabled: true
    showProgress: true
    
  # Navigation
  toc:
    enabled: true
    position: "right"
    maxDepth: 3
    
  # Metadata
  reading-time:
    enabled: true
    wordsPerMinute: 200
    
  # Engagement
  view-counter:
    enabled: false
    provider: "local"
    
  # External
  external-links:
    enabled: true
    icon: "↗"
    openInNewTab: true
```
